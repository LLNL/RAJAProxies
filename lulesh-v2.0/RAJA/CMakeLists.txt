set (lulesh_sources
  lulesh.cpp
  lulesh-comm.cpp
  lulesh-init.cpp
  lulesh-util.cpp
  lulesh-viz.cpp)

set (lulesh_depends
  RAJA)

if (ENABLE_OPENMP)
  set (lulesh_depends
    ${lulesh_depends}
    openmp)
endif ()

if (ENABLE_CUDA)
  set (lulesh_depends
    ${lulesh_depends}
    cuda)
endif ()

if (ENABLE_CHAI)
  set (lulesh_depends
    ${lulesh_depends}
    chai)
endif ()

blt_add_executable(
  NAME lulesh-v2.0-RAJA-seq.exe
  SOURCES ${lulesh_sources}
  DEPENDS_ON ${lulesh_depends})

target_compile_options(
  lulesh-v2.0-RAJA-seq.exe
  PRIVATE
  -Wno-unknown-pragmas)

target_compile_definitions(
  lulesh-v2.0-RAJA-seq.exe
  PRIVATE 
  -DUSE_MPI=0)

target_compile_definitions(
  lulesh-v2.0-RAJA-seq.exe
  PRIVATE
  -DUSE_CASE=1)

target_compile_definitions(
  lulesh-v2.0-RAJA-seq.exe
  PUBLIC
  -DLULESH_DEVICE= 
  -DLULESH_LAMBDA_CLOSURE_ATTR=
  -DLULESH_LAMBDA_DEFAULT_CAPTURE_MODE=& 
)

if (ENABLE_OPENMP)
  blt_add_executable(
    NAME lulesh-v2.0-RAJA-omp.exe
    SOURCES ${lulesh_sources}
    DEPENDS_ON ${lulesh_depends})

  target_compile_definitions(
    lulesh-v2.0-RAJA-omp.exe
    PRIVATE 
    -DUSE_MPI=0)

  target_compile_definitions(
    lulesh-v2.0-RAJA-omp.exe
    PUBLIC
    -DLULESH_DEVICE= 
    -DLULESH_LAMBDA_CLOSURE_ATTR=
    -DLULESH_LAMBDA_DEFAULT_CAPTURE_MODE=& 
  )
endif ()

if (ENABLE_CUDA)
  configure_file(
    lulesh.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/lulesh-cuda.cpp COPYONLY)

  configure_file(
    lulesh-comm.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/lulesh-comm-cuda.cpp COPYONLY)
  configure_file(
    lulesh-init.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/lulesh-init-cuda.cpp COPYONLY)
  configure_file(
    lulesh-util.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/lulesh-util-cuda.cpp COPYONLY)
  configure_file(
    lulesh-viz.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/lulesh-viz-cuda.cpp COPYONLY)

  blt_add_executable(
    NAME lulesh-v2.0-RAJA-cuda.exe
    SOURCES 
    ${CMAKE_CURRENT_BINARY_DIR}/lulesh-cuda.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/lulesh-comm-cuda.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/lulesh-init-cuda.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/lulesh-util-cuda.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/lulesh-viz-cuda.cpp
    DEPENDS_ON ${lulesh_depends})

  target_compile_definitions(
    lulesh-v2.0-RAJA-cuda.exe
    PRIVATE
    -DUSE_CASE=9)

  target_compile_definitions(
    lulesh-v2.0-RAJA-cuda.exe
    PRIVATE 
    -DUSE_MPI=0)

  target_compile_definitions(
    lulesh-v2.0-RAJA-cuda.exe
    PUBLIC
    -DLULESH_DEVICE=__device__
    -DLULESH_LAMBDA_CLOSURE_ATTR=mutable 
    -DLULESH_LAMBDA_DEFAULT_CAPTURE_MODE== 
  )

  target_include_directories(
    lulesh-v2.0-RAJA-cuda.exe
    PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR})
endif ()
